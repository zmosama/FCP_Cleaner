سكربت لمسح الفولدرات الكبيرة من مشاريع Final Cut Pro
قبل عمل الباك أب لتوفير المساحة

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
📁 الملفات المتاحة
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

استخدم الـ Double-Click على الملفات التالية:

🔍 معاينة_فقط.command
   → شوف التقرير والمساحة القابلة للمسح (بدون مسح فعلي)
   → آمن 100% - مجرد معاينة

🗑️ مسح_تلقائي.command
   → امسح كل الفولدرات الكبيرة مرة واحدة
   → هيطلب تأكيد قبل المسح

⚙️ تفاعلي.command
   → اختار بنفسك إيه اللي عايز تمسحه
   → تحكم كامل في عملية المسح

✅ الفولدرات اللي بتتمسح
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

السكربت بيمسح الفولدرات دي فقط:

  ✓ Analysis Files      (ملفات التحليل)
  ✓ Render Files        (ملفات الريندر)
  ✓ Transcoded Media    (الميديا المحولة)

هذه الملفات تتعاد إنشاءها تلقائياً لما تفتح المشروع
في Final Cut Pro مرة تانية

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
❌ الملفات اللي مش بتتمسح
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

السكربت بيحافظ على:

  ✓ Original Media           (الملفات الأصلية)
  ✓ CurrentVersion.fcpevent  (المشروع نفسه)
  ✓ أي ملفات تانية مهمة

مشروعك هيفضل سليم 100%!

🚀 الاستخدام السريع
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

الخطوات:

1. افتح فولدر backup-cleaner
2. اعمل Double-Click على: 🔍_معاينة_فقط.command
3. شوف التقرير والمساحة اللي هتوفرها
4. لو عجبك، اعمل Double-Click على: 🗑️_مسح_تلقائي.command
5. انتهى! وفرت 50-70% من المساحة
